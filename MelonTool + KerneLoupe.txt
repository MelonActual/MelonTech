    Generates MT+KL inventory items (2004-2018) with period-accurate specs.
    Returns formatted dictionary matching existing store entries.

        2004: {"aesthetic": "Pre-iPhone utilitarian", "tech": "PDA-era"},
        2008: {"aesthetic": "Early smartphone fusion", "tech": "Capacitive touch dawn"},
        2012: {"aesthetic": "Post-industrial minimalism", "tech": "ARM revolution"},
        2016: {"aesthetic": "Cyberpunk revival", "tech": "USB-C transition"}
    }

    # --- MATERIAL LIBRARY ---
    materials = {
        'plastics': ["ABS", "Polycarbonate", "Kydex"],
